The following component will only use [`<OffthreadVideo />`](/docs/offthreadvideo) while rendering, but [`<Video />`](/docs/video) in the Player.
This is useful for attaching a `ref` to the [`<Video />`](/docs/video) tag.


```tsx twoslash
import { forwardRef } from 'react';
import {
  getRemotionEnvironment,
  OffthreadVideo,
  RemotionOffthreadVideoProps,
  Video,
} from 'remotion';

function OffthreadWhileRenderingRefForwardingFunction(
  props: RemotionOffthreadVideoProps,
  ref: React.Ref<HTMLVideoElement>,
) {
  const { onAutoPlayError, ...otherProps } = props;
  const isPreview = !getRemotionEnvironment().isRendering;

  const onAutoPlayErrorNotNull = onAutoPlayError ?? undefined;
  if (isPreview) {
    return <Video ref={ref} onAutoPlayError={onAutoPlayErrorNotNull} {...otherProps} />;
  }

  return <OffthreadVideo onAutoPlayError={onAutoPlayError} {...props} />;
}

export const OffthreadVideoWhileRendering = forwardRef(
  OffthreadWhileRenderingRefForwardingFunction,
);
```
