## Example usage


```tsx title="app.tsx"

import {  downloadWhisperModel, transcribe } from "@remotion/whisper-wasm"
import { useEffect, useState } from "react";



export default function Home() {
  const [downloadProgress, setDownloadProgress] = useState(0);
  const [transcriptionProgress, setTranscriptionProgress] = useState(0);
  const [text, setText] = useState("");

  useEffect(()=>{
    downloadWhisperModel({model:'tiny.en', onProgress: downloadProgress});
  },[]);

  const onTranscribeChunk = (start:string, end:string, text:string) => {
    // do something
  }

  return (
     <div>
    
            <input type="file" accept=".wav" onChange={(e)=>{
            transcribe({file: e.target.files[0], onProgress: transcriptionProgress, onTranscribeChunk, threads: 12})
                 .then(result => setText(result));

   
            }} />
            
      <progress value={transcriptionProgress} max={100} />
      </div>
  );
}

```
