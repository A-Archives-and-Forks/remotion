## `MediaParserVideoTrack`

```tsx twoslash
import {MediaParserVideoCodec, MediaParserCodecData, MediaParserSampleAspectRatio, VideoTrackColorParams} from '@remotion/media-parser';

export type MediaParserVideoTrack = {
  // WebCodecs
  codec: string;
  description: Uint8Array | undefined;
  codedWidth: number;
  codedHeight: number;
  displayAspectWidth: number;
  displayAspectHeight: number;
  // Non-WebCodecs
  type: 'video';
  trackId: number;
  codecEnum: MediaParserVideoCodec;
  codecData: MediaParserCodecData | null;
  sampleAspectRatio: MediaParserSampleAspectRatio;
  width: number;
  height: number;
  rotation: number;
  fps: number | null;
  timescale: number;
  color: VideoTrackColorParams;
  m3uStreamFormat: 'ts' | 'mp4' | null;
};
```
